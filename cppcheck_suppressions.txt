// S.A.K. Utility — cppcheck suppressions
// Only fundamental tool limitations. All real findings fixed in code.

// cppcheck does not have Qt/system headers available.
missingInclude
missingIncludeSystem

// Single-file analysis cannot determine cross-TU function/member usage.
unusedFunction
unusedStructMember

// Meta-noise from the suppression file itself.
unmatchedSuppression

// cppcheck does not understand Qt macros even with -D defines.
unknownMacro

// Q_EMIT signal(...) is misidentified as a variable shadowing the signal decl.
shadowFunction

// Style suggestion to use STL algorithms; Qt-idiomatic loops are preferred.
useStlAlgorithm

// Qt Test slots are invoked by the meta-object system and cannot be static.
functionStatic

// Test assertions intentionally verify enum distinctness with known values.
knownConditionTrueFalse
